SELECT ?player ?playerLabel ?wikipediaArticle ?award ?awardLabel
WHERE {
  # Buscar jugadores de fútbol (profesionales) con la ocupación de futbolista (P106) y el nombre 'Iniesta'
  ?player wdt:P106 wd:Q937857;  # P106: ocupación (jugador de fútbol), Q937857: futbolista
          rdfs:label ?playerLabel.
  
  # Filtro por nombre (en este caso 'Iniesta')
  FILTER(CONTAINS(LCASE(?playerLabel), "iniesta")).
  FILTER(LANG(?playerLabel) = "en").  # Limitar a etiquetas en inglés

  # Obtener el artículo de Wikipedia en inglés
  OPTIONAL {
    ?wikipediaArticle schema:about ?player;
                      schema:isPartOf <https://en.wikipedia.org/>.
  }
# Premios ganados
  OPTIONAL { 
    ?player wdt:P166 ?award.  # Premios (P166)
    ?award rdfs:label ?awardLabel FILTER( LANG(?awardLabel) = "en").
  }
  # Limitar las etiquetas en inglés
  SERVICE wikibase:label { bd:serviceParam wikibase:language "en". }
}
LIMIT 10
